Komenda autoprefixer-cli do czytania z pliku style.css i zapisywania wyniku do pliku style.prefixed.style:

autoprefixer-cli -o style.prefixed.css style.css




Komnenda nu-html-checker do wyswietlania listy błędów we wszystkich plikach .html

var checker = require('nu-html-checker');
 
var batch = checker.validate([ '.html' ]);
 
batch.on('progress', function (e) {
  console.log(checker.format('stylish', e.value), '\n');
});
 
batch.on('end', function (err) {
  if (err) console.error(err.stack);
})
